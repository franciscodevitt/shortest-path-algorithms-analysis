@startuml

' ====== Clases externas usadas ======
class Comparador
class Lista
class Tupla

' ====== Clase Nodo ======
class Nodo<C,V> {
    C clave
    V valor
    Nodo padre
    Nodo hijoIzquierdo
    Nodo hijoDerecho

    + Nodo(clave : C, valor : V, padre : Nodo, hijoIzq : Nodo, hijoDer : Nodo)
    + Nodo(clave : C, valor : V, padre : Nodo)
    + Nodo(clave : C, valor : V)
}

' ====== Clase DiccionarioOrdenado ======
class DiccionarioOrdenado<C,V> {
    - Nodo raiz
    - Comparador comparador
    - int cantidadDatos

    + DiccionarioOrdenado(comparador : Comparador)
    + DiccionarioOrdenado(diccionario : DiccionarioOrdenado)

    + V agregar(clave : C, valor : V)
    + V eliminar(clave : C)
    + V obtenerValor(clave : C)

    + Lista inorder()
    + Lista preorder()
    + Lista postorder()
    + Lista ancho()

    + int tamanio()
    + boolean vacio()

    -- Métodos privados --
    - Nodo obtenerSucesorInmediato(nodo : Nodo)
    - void inorderRecorrer(nodo : Nodo, lista : Lista)
    - void preorderRecorrer(nodo : Nodo, lista : Lista)
    - void postorderRecorrer(nodo : Nodo, lista : Lista)
    - Nodo clonarSubArbol(nodo : Nodo)
    - int cmp(clave1 : C, clave2 : C)
    - Nodo buscarNodo(clave : C, raiz : Nodo)
    - int cantHijos(nodo : Nodo)
    - void borrarNodoConUnHijo(nodo : Nodo, padre : Nodo)
}

' ====== Relaciones ======

' El diccionario tiene una raíz (composición)
DiccionarioOrdenado *-- Nodo : raiz

' Cada nodo tiene referencias a sus hijos
Nodo --> Nodo : hijoIzquierdo
Nodo --> Nodo : hijoDerecho
Nodo --> Nodo : padre

DiccionarioOrdenado --> Comparador : usa
DiccionarioOrdenado --> Lista : retorna listas
DiccionarioOrdenado --> Tupla : elementos (C,V)
DiccionarioOrdenado --> Cola : recorrido en ancho

@enduml
